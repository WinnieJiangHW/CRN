from torch import nn
from TCN.tcn import TemporalConvNet
import torch.nn.functional as F
import torch as t


class CRN(nn.Module):
    def __init__(self, input_size, output_size, num_channels, kernel_size, dropout):
        super(CRN, self).__init__()
        self.tcn = TemporalConvNet(input_size, num_channels, kernel_size, dropout=dropout)
        self.cell = nn.RNNCell(input_size, num_channels[-1])
        self.lstm = nn.LSTM(num_channels[-1], num_channels[-1])
        self.linear = nn.Linear(num_channels[-1], output_size)
        self.sig = nn.Sigmoid()

    def forward(self, x):
        # x needs to have dimension (N, C, L) in order to be passed into CNN
        output = self.tcn(x.transpose(1, 2)).transpose(1, 2)
        output = self.cell(x.view(-1, x.shape[2]), output.view(-1, output.shape[2])).view(1, output.shape[1], -1)
        h0 = t.randn(1, x.shape[0], output.shape[2]).cuda()
        c0 = t.randn(1, x.shape[0], output.shape[2]).cuda()
        output, (hn, cn) = self.lstm(output.transpose(0, 1), (h0, c0))
        output = output.transpose(0, 1)
        output = self.linear(output).double()
        return self.sig(output)
